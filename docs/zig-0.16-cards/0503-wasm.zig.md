## 3) The Golden Snippet

```zig
const std = @import("std");
const wasm = std.wasm;

// Using WebAssembly opcodes and types
pub fn main() void {
    const opcode = wasm.Opcode.i32_const;
    const valtype = wasm.Valtype.i32;
    const block_type = wasm.BlockType.fromValtype(valtype);
    
    std.debug.print("Opcode: {}\n", .{@intFromEnum(opcode)});
    std.debug.print("Value type: {}\n", .{@intFromEnum(valtype)});
    std.debug.print("Block type: {}\n", .{@intFromEnum(block_type)});
    
    // Check module magic and version
    std.debug.print("Wasm magic: {x}\n", .{wasm.magic});
    std.debug.print("Wasm version: {x}\n", .{wasm.version});
}
```
