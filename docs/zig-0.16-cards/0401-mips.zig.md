## 3) The Golden Snippet

const std = @import("std");
const mips = std.os.linux.mips;

// Example: Using syscall3 to call read system call
const SYS = std.os.linux.SYS;

pub fn example_read(fd: i32, buffer: []u8) u32 {
    return mips.syscall3(
        SYS.read,
        @as(u32, @bitCast(fd)),
        @intFromPtr(buffer.ptr),
        @as(u32, buffer.len)
    );
}
